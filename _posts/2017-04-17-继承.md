## 原型继承

```js
	// 人类 → 父类
    function Person() {
      this.name = '名字';
      this.age = 10;
      this.gender = '男';
    }
    Person.prototype.sayHi = function () {
      console.log('hello');
    }
    Person.prototype.run = function () {
      console.log('我会跑....')
    }
    Person.prototype.eat = function () {
      console.log('吃吃吃....')
    }
     // 学生类 → 子类
    function Student() {
      this.stuId = 10010;
    }
    // 如何实现原型继承
    // 利用原型链的特点
    // 1. 把子类的prototype指向父类的一个实例
    Student.prototype = new Person();
	// 2. 给原型设置一个constructor属性指向构造函数
    Student.prototype.constructor = Student;
	// 创建对象
    var stu1 = new Student();
```

```js
	// 问题：无法应对属性的变化
    // 原因：stu1和stu2共享一个原型
    // 解决：借用继承

    // 原型继承优缺点：
    // 优点：方法可以完美的继承
    // 缺点：无法完美继承属性→属性是变化时
```

## 借用继承

```js
	// 主题：研究this指向
    // 要在函数内部研究
    // 一个函数内部this到底指向谁？不是在函数定义时决定定，而是在函数调用并执行时决定的。
```

```js
	function Person(name,age,gender) {
      this.name = name;
      this.age = age;
      this.gender = gender;
    }
    // ① 普通函数调用时：this→window
    // Person('张三',10,'男');
    // window.Person('张三',10,'男');

    // ② 改变函数内部this指向
    // 函数名.call(调用者,函数实参，函数的实参...)
    var obj = {};  // 
    Person.call(obj,'张三',10,'男')
```

```js
	// 人类 → 父类
    function Person(name, age, gender) {
      this.name = name;
      this.age = age;
      this.gender = gender;
    }
    Person.prototype.sayHi = function () {
      console.log('hello');
    }
    Person.prototype.run = function () {
      console.log('我会跑....')
    }
    Person.prototype.eat = function () {
      console.log('吃吃吃....')
    }

    // 学生类 → 子类
    function Student(name, age, gender) {
      this.stuId = 10010;
      // this 代表当前创建的对象，stu1,stu2
      Person.call(this, name, age, gender)
    }

    // 创建对象stu1
    var stu1 = new Student('张三', 10, '女');
    // 创建对象stu2
    var stu2 = new Student('李四', 18, '男');
```

```js
	// 优点：完美继承了属性
    // 缺点：无法实现继承方法
```

## 组合继承

```js
	// 人类 → 父类
    function Person(name, age, gender) {
      this.name = name;
      this.age = age;
      this.gender = gender;
    }
    Person.prototype.sayHi = function () {
      console.log('hello');
    }
    Person.prototype.run = function () {
      console.log('我会跑....')
    }
    Person.prototype.eat = function () {
      console.log('吃吃吃....')
    }
    
    // 学生类 → 子类
    function Student(name, age, gender) {
      this.stuId = 10010;
      // 【借用继承】
      // this 代表当前创建的对象，stu1,stu2
      Person.call(this, name, age, gender)
    }
    // 【原型继承】
    Student.prototype = new Person();
    Student.prototype.constructor = Student;

    // 创建对象stu1
    var stu1 = new Student('张三', 10, '女');
    // 创建对象stu2
    var stu2 = new Student('李四', 18, '男');
```

```js
	// 借用：
    // 优点：完美继承了属性
    // 缺点：无法实现继承方法

    // 原型继承优缺点：
    // 优点：方法可以完美的继承
    // 缺点：无法完美继承属性→属性是变化时
```

## 寄生式继承

> 寄生式继承的思路与(寄生)构造函数和工厂模式类似, 即创建一个仅用于封装继承过程的函数,该函数在内部以某种方式来增强对象,最后再像真的是它做了所有工作一样返回对象. 如下.

```js
function createAnother(original){
    var clone = object(original);//通过调用object函数创建一个新对象
    clone.sayHi = function(){//以某种方式来增强这个对象
        alert("hi");
    };
    return clone;//返回这个对象
}
```

## 寄生组合式继承

组合继承是 JavaScript 最常用的继承模式; 不过, 它也有自己的不足. 组合继承最大的问题就是无论什么情况下,都会调用两次父类构造函数: 一次是在创建子类型原型的时候, 另一次是在子类型构造函数内部. **寄生组合式继承就是为了降低调用父类构造函数的开销而出现的** .

> 其背后的基本思路是: 不必为了指定子类型的原型而调用超类型的构造函数

```js
function extend(subClass,superClass){
    var prototype = object(superClass.prototype);//创建对象
    prototype.constructor = subClass;//增强对象
    subClass.prototype = prototype;//指定对象
}
```

extend的高效率体现在它没有调用superClass构造函数,因此避免了在subClass.prototype上面创建不必要,多余的属性. 于此同时,原型链还能保持不变; 因此还能正常使用 instanceof 和 isPrototypeOf() 方法.

以上,寄生组合式继承,集寄生式继承和组合继承的优点于一身,是实现基于类型继承的最有效方法.
