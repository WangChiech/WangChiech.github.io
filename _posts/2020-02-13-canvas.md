canvas 是HTML提供的新标签，矩形画布，可以用 JavaScript 控制每一像素在上面进行绘画，拥有绘制路径、矩形、圆形等。

canvas 可以应用于游戏、可视化数据（数据图表，如百度的echar)、banner 广告、未来可能会有模拟器（航天航空等）、远程计算机控制、图形编辑器（Photoshop网页版）、完整的canvas移动化应用.....

```html
<!-- 设置canvas的宽高是通过canvas的内部属性来设置的，通过css设置的宽高会拉伸canvas -->
  <canvas id="demo" width="600" height="600">
    ie9 以上才支持
    你的浏览器不支持 canvas
    <!-- flash -->
  </canvas>
  <script>
    // 1. 拿到canvas标签
    var canvas = document.getElementById('demo');
    canvas.style.border = '1px solid red';
    // canvas.width = 600; 
    // canvas.height = 600;
    // canvas.style.height = 1200 // 错误，不使用css设置canvas样式

    // 2. 拿到canvas的上下文
    var context = canvas.getContext('2d'); // 'webgl'

    // 绘制
    context.moveTo(0,0); // 画笔移动到（100,100）
    context.lineTo(200, 100); 
    context.lineTo(100, 200);
    // context.lineTo(0,0) 
    context.closePath() //闭合标签
    
    // 设置线宽
    context.lineWidth = 4;

    context.strokeStyle = 'red';  // 描边颜色
    context.stroke(); // 描边

    context.fillStyle = 'green'; // 需在context.fill()前设置才能生效
    context.fill(); // 可以自动闭合路径
  </script>
```
