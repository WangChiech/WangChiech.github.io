### 18 循环于事件结合的问题

```html
<ul>
  <li>1</li>
  <li>2</li>
  <li>3</li>
  <li>4</li>
  <li>5</li>
  <li>6</li>
</ul>
<!-- 如何获取每个li -->
<script>
  // 错误， 每次点击都是 6
  const lis = document.querySelectorAll('li')
  for (var i = 0; i < lis.length; i++) {
    lis[i].onclick = function() {
      console.log(i)
    }
  }
  // 1.每次点击为 li 对应索引值
  for (let i = 0; i < lis.length; i++) {
    lis[i].onclick = function() {
      console.log(i)
    }
  }
  // 2.自定义属性，每次点击为 li 对应索引值
  for (let i = 0; i < lis.length; i++) {
    lis[i].index = i
    lis[i].onclick = function() {
      console.log(this.index)
    }
  }
  // 3.自执行函数，每次点击为 li 对应索引值
  for (let i = 0; i < lis.length; i++) {
    (function(i) {
       lis[i].onclick = function() {
      		console.log(this.index)
    	} 
    })(i) 
  }
  // 4.数组forEach、map、filter，每次点击为 li 对应索引值
  const newLis = [].slice.call(lis)
  newLis.forEach(function(value, index, arr) {
      value.onclick = function() {
          console.log(index)
      }
  })
  // 5.Array.from，每次点击为 li 对应索引值
  Array.from(lis, function(value, index) {
      value.onclick = function() {
          console.log(index)
      }
  })


</script>
```
